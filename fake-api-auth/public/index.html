You need to authenticate at POST /auth/login to get access_token

    
    fetch("http://localhost:4000/auth/login", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        login,
        password,
      }),
    })

Here'are some examples what you might want to do :

  • list all todos: GET /todos
  • add a todo POST /todos
  • update a todo PUT /todos/1
  • delete a todo DELETE /todos/1

fetch("http://localhost:4000/todos", {
  headers: {
    Authorization: "Bearer " + user.access_token,
  }
})



fetch("http://localhost:4000/todos", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    Authorization: "Bearer " + user.access_token,
  },
  body: JSON.stringify({
    text: "Add a new todo",
    isCompleted: false
  })
})



fetch("http://localhost:4000/todos/5", {
  method: "PUT",
  headers: {
    "Content-Type": "application/json",
    Authorization: "Bearer " + user.access_token,
  },
  body: JSON.stringify({
    text: "Add new todo via API",
    isCompleted: true,
  }),
})



fetch("http://localhost:4000/todos/5", {
  method: "DELETE",
  headers: {
    "Content-Type": "application/json",
    Authorization: "Bearer " + user.access_token,
  },
})
